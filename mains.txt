void	ft_lstclear(t_list **lst, void (*del)(void *))
{
	t_list	*temp;
	t_list	*tmpnxt;

	if (!lst || !del)
		return ;
	temp = *lst;
	while (temp != NULL)
	{
		tmpnxt = temp->next;
		ft_lstdelone(temp, del);
		temp = tmpnxt;
	}
	*lst = NULL;
}

t_list	*ft_lstmap(t_list *lst, void *(*f)(void *), void (*del)(void *))
{
	t_list	*newlst;
	t_list	*temp;

	if (!lst || !f || !del)
		return (NULL);
	temp = lst;
	newlst = 0;
	while (temp)
	{
		ft_lstadd_back(&newlst, ft_lstnew(f(temp->content)));
		if (!newlst)
		{
			ft_lstclear(&lst, del);
			return (NULL);
		}
		temp = temp->next;
	}
	return (newlst);
}

char	*ft_skfkf(char *s)
{
	int i = 0;
	char * news = ft_strdup(s);
	while (s[i])
	{
		news[i] = ft_toupper(news[i]);
		i++;
	}
	return (news);
}
